    Step 6 — Create src/model/transforms.ts with `evaluateWorldTransform`:

    ```typescript
    import type { Bone, BoneTransform, Skeleton } from './types'

    export interface WorldTransform extends BoneTransform {}

    export function evaluateWorldTransform(boneId: string, skeleton: Skeleton): WorldTransform {
      const bone = skeleton.bones[boneId]
      if (!bone) throw new Error(`Bone not found: ${boneId}`)

      if (!bone.parentId) {
        // Root bone — local IS world
        return { ...bone.localTransform }
      }

      const parentWorld = evaluateWorldTransform(bone.parentId, skeleton)
      const local = bone.localTransform

      // Apply parent's rotation to the child's local position offset
      const cos = Math.cos(parentWorld.rotation)
      const sin = Math.sin(parentWorld.rotation)
      const scaledX = local.x * parentWorld.scaleX
      const scaledY = local.y * parentWorld.scaleY

      return {
        x: parentWorld.x + cos * scaledX - sin * scaledY,
        y: parentWorld.y + sin * scaledX + cos * scaledY,
        rotation: parentWorld.rotation + local.rotation,
        scaleX: parentWorld.scaleX * local.scaleX,
        scaleY: parentWorld.scaleY * local.scaleY,
      }
    }
    ```

    CRITICAL NOTE from research (Open Question #2): Do NOT use the Zustand `immer` middleware for document mutations. The `immer` middleware wraps `set` with `produce` (not `produceWithPatches`), which does not emit patches. Use the `immer` middleware ONLY for view state mutations (selection, viewport). Use raw `set` + explicit `produceWithPatches` for document mutations.

    Step 1 — Create src/store/undoRedo.ts:

    ```typescript
    import { enablePatches, produceWithPatches, applyPatches, Patch } from 'immer'
    import type { Skeleton, ImageAsset, Attachment } from '../model/types'

    // MUST be called once before any produce call. Place here so it runs at module load.
    enablePatches()

    export type PatchEntry = { inverse: Patch[]; forward: Patch[] }

    /** Wraps a document mutation to capture Immer patches for undo/redo. */
    export function withUndo<T extends object>(
      current: T,
      recipe: (draft: T) => void,
      undoStack: PatchEntry[],
      redoStack: PatchEntry[],
      maxStack = 100
    ): { next: T; undoStack: PatchEntry[]; redoStack: PatchEntry[] } {
      const [next, patches, inversePatches] = produceWithPatches(current, recipe)
      const entry: PatchEntry = { inverse: inversePatches, forward: patches }
      const newUndo = [...undoStack, entry]
      if (newUndo.length > maxStack) newUndo.shift()
      return { next, undoStack: newUndo, redoStack: [] } // new mutation clears redo
    }

    export { applyPatches }
    ```

    Actions to implement:
    - `createBone(parentId: string | null): string` — creates bone with default name `Bone_N`, generates UUID via `crypto.randomUUID()`, adds to parent's `childIds`, adds to `rootBoneIds` if root. Returns the new bone ID.

    - `deleteBone(boneId: string): void` — re-parents children to deleted bone's parent (or makes them root). Removes from parent's `childIds` and `rootBoneIds`.
    - `reparentBone(boneId: string, newParentId: string | null): void` — removes from old parent's `childIds`/`rootBoneIds`, adds to new parent's `childIds`/`newRootBoneIds`.
    - `setBoneTransform(boneId: string, transform: Partial<import('../../model/types').BoneTransform>): void` — merges into `bone.localTransform`; also updates `bindTransform` when `editorMode === 'pose'`.
    - `setBoneVisibility(boneId: string, visible: boolean): void`
    - `importImage(asset: import('../../model/types').ImageAsset): void`
    - `attachImage(attachment: import('../../model/types').Attachment): void`
    - `detachImage(attachmentId: string): void`
    - `updateAttachment(attachmentId: string, patch: Partial<import('../../model/types').Attachment>): void`
    - `undo(): void` — pops from undoStack, applies inverse patches to skeleton, pushes to redoStack
    - `redo(): void` — pops from redoStack, applies forward patches to skeleton, pushes to undoStack

    For undo/redo: apply patches to `skeleton` only (Phase 1 document is just skeleton + imageAssets + attachments). Structure:

    ```typescript
    undo: () => set((state) => {
      const entry = state.undoStack.pop()
      if (!entry) return
      state.skeleton = applyPatches(state.skeleton as any, entry.inverse) as Skeleton
      state.redoStack.push(entry)
    }),
    ```
