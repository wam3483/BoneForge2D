  <name>Task 1: Move and Rotate gizmos with globalpointermove drag</name>
  <files>src/viewport/GizmoLayer.ts</files>
  <action>
    CRITICAL from research:
    - ALWAYS use `app.stage.on('globalpointermove', ...)` for drag — NOT `pointermove` on the gizmo handle. The gizmo handle is small; pointer exits immediately on fast drag. `globalpointermove` fires regardless of pointer position.
    - Remove `globalpointermove` and `pointerup` listeners when drag ends.
    - Gizmos show ONLY when a bone is selected AND the matching tool is active.
    - Gizmo transforms are in LOCAL SPACE — convert screen deltas to local-space deltas using the parent's world transform.

    Create src/viewport/GizmoLayer.ts as an imperative class instantiated by BoneRendererLayer (or PixiViewport). The class:
    - Is given the `app`, the camera's `bonesContainer`, and `screenToWorld` access
    - Subscribes to Zustand store state changes to show/hide/reposition gizmos
    - Uses `useTick` to reposition gizmo handles each frame to track the selected bone

    **GizmoLayer class structure:**

    ```typescript
    import { Application, Container, Graphics, FederatedPointerEvent, Point } from 'pixi.js'
    import { useEditorStore } from '../store'
    import { evaluateWorldTransform } from '../model/transforms'
    import type { BoneTransform, Skeleton } from '../model/types'

    type GizmoMode = 'move' | 'rotate' | 'scale' | null

    export class GizmoLayer {
      private app: Application
      private cameraContainer: Container
      private gizmoContainer: Container
      private moveGizmo: MoveGizmo
      private rotateGizmo: RotateGizmo
      private scaleGizmo: ScaleGizmo
      private activeGizmo: GizmoMode = null
      private isDragging = false
      private dragStart = { screenX: 0, screenY: 0, localX: 0, localY: 0 }
      private boneStartTransform: BoneTransform | null = null

      constructor(app: Application, cameraContainer: Container) {
        this.app = app
        this.cameraContainer = cameraContainer
        this.gizmoContainer = new Container()
        cameraContainer.addChild(this.gizmoContainer)
        this.moveGizmo = new MoveGizmo(this.gizmoContainer)
        this.rotateGizmo = new RotateGizmo(this.gizmoContainer)
        this.scaleGizmo = new ScaleGizmo(this.gizmoContainer)
        this.setupDragHandlers()
      }

      private setupDragHandlers(): void {
        const stage = this.app.stage

        stage.on('globalpointermove', (e: FederatedPointerEvent) => {
          if (!this.isDragging || !this.activeGizmo) return
          this.applyDrag(e)
        })
        stage.on('pointerup', () => this.endDrag())
        stage.on('pointerupoutside', () => this.endDrag())
      }

      // Called from useTick each frame
      update(): void {
        const state = useEditorStore.getState()
        const { selectedBoneId, skeleton, activeTool } = state
        const tool = activeTool === 'select' ? null : activeTool as GizmoMode

        // Hide all gizmos if nothing selected or wrong tool
        if (!selectedBoneId || !skeleton.bones[selectedBoneId] || !tool) {
          this.moveGizmo.hide(); this.rotateGizmo.hide(); this.scaleGizmo.hide()
          return
        }

        const world = evaluateWorldTransform(selectedBoneId, skeleton)
        this.moveGizmo.setVisible(tool === 'move')
        this.rotateGizmo.setVisible(tool === 'rotate')
        this.scaleGizmo.setVisible(tool === 'scale')
        this.moveGizmo.position(world.x, world.y, world.rotation)
        this.rotateGizmo.position(world.x, world.y, world.rotation)
        this.scaleGizmo.position(world.x, world.y, world.rotation)
      }

      startDrag(gizmoMode: GizmoMode, e: FederatedPointerEvent, boneId: string): void {
        const state = useEditorStore.getState()
        this.isDragging = true
        this.activeGizmo = gizmoMode
        this.boneStartTransform = { ...state.skeleton.bones[boneId].localTransform }
        this.dragStart = { screenX: e.global.x, screenY: e.global.y, localX: 0, localY: 0 }
      }

      private endDrag(): void {
        this.isDragging = false
        this.activeGizmo = null
        this.boneStartTransform = null
      }

      private applyDrag(e: FederatedPointerEvent): void {
        const state = useEditorStore.getState()
        const { selectedBoneId, skeleton, snapEnabled, snapGridSize } = state
        if (!selectedBoneId || !this.boneStartTransform) return

        const bone = skeleton.bones[selectedBoneId]
        const snap = (v: number) => snapEnabled ? Math.round(v / snapGridSize) * snapGridSize : v

        if (this.activeGizmo === 'move') {
          // Convert screen delta to world delta (divided by camera scale)
          const camScale = this.cameraContainer.scale.x
          let dx = (e.global.x - this.dragStart.screenX) / camScale
          let dy = (e.global.y - this.dragStart.screenY) / camScale

          // Convert world delta to local delta by rotating by negative parent rotation
          const parentWorldRot = bone.parentId
            ? evaluateWorldTransform(bone.parentId, skeleton).rotation
            : 0
          const cos = Math.cos(-parentWorldRot)
          const sin = Math.sin(-parentWorldRot)
          const localDx = snap(cos * dx - sin * dy)
          const localDy = snap(sin * dx + cos * dy)

          useEditorStore.getState().setBoneTransform(selectedBoneId, {
            x: this.boneStartTransform.x + localDx,
            y: this.boneStartTransform.y + localDy,
          })
        }

        if (this.activeGizmo === 'rotate') {
          const world = evaluateWorldTransform(selectedBoneId, skeleton)
          const camScale = this.cameraContainer.scale.x
          // World position of bone origin in screen space
          const boneScreenX = world.x * camScale + this.cameraContainer.x
          const boneScreenY = world.y * camScale + this.cameraContainer.y
          const startAngle = Math.atan2(this.dragStart.screenY - boneScreenY, this.dragStart.screenX - boneScreenX)
          const currentAngle = Math.atan2(e.global.y - boneScreenY, e.global.x - boneScreenX)
          const deltaAngle = currentAngle - startAngle

          useEditorStore.getState().setBoneTransform(selectedBoneId, {
            rotation: this.boneStartTransform.rotation + deltaAngle,
          })
        }

        if (this.activeGizmo === 'scale') {
          const world = evaluateWorldTransform(selectedBoneId, skeleton)
          const camScale = this.cameraContainer.scale.x
          const boneScreenX = world.x * camScale + this.cameraContainer.x
          const boneScreenY = world.y * camScale + this.cameraContainer.y
          const startDist = Math.hypot(this.dragStart.screenX - boneScreenX, this.dragStart.screenY - boneScreenY)
          const currentDist = Math.hypot(e.global.x - boneScreenX, e.global.y - boneScreenY)
          if (startDist < 1) return
          const scaleFactor = Math.max(0.01, currentDist / startDist)

          useEditorStore.getState().setBoneTransform(selectedBoneId, {
            scaleX: this.boneStartTransform.scaleX * scaleFactor,
            scaleY: this.boneStartTransform.scaleY * scaleFactor,
          })
        }
      }

      destroy(): void {
        this.gizmoContainer.destroy({ children: true })
      }
    }
    ```

    **MoveGizmo class** (nested in same file):
    - Three handles: X axis arrow (red, horizontal), Y axis arrow (green, vertical), center square (white, free-move)
    - Each handle is a Graphics with `eventMode = 'static'`
    - On `pointerdown`, calls `gizmoLayer.startDrag('move', e, selectedBoneId)` with e.stopPropagation()
    - Renders X arrow as horizontal line with arrowhead at (40, 0); Y arrow as vertical line with arrowhead at (0, -40); center as 8x8 square at (0, 0)
    - `position(x, y, rotation)`: sets container position, applies rotation

    **RotateGizmo class** (nested in same file):
    - Single circular arc handle: draw arc at radius 50px around bone origin, arc from -45deg to 180deg (3/4 circle)
    - `eventMode = 'static'`, `hitArea` is an Annulus approximation (use full circle at radius 50 with line width 12 for hit area)
    - On `pointerdown` at start of arc, records start pointer angle; `startDrag('rotate', ...)`
    - Small angle indicator arrow at the arc endpoint

    **ScaleGizmo class** (nested in same file):
    - Two handles: X scale (horizontal line at 50px, square endpoint) and Y scale (vertical line at 50px, square endpoint)
    - Drag along axis increases scale; start distance vs current distance ratio = scale factor
    - On `pointerdown`, `startDrag('scale', ...)`

    ```typescript
    import { Application, FederatedPointerEvent } from 'pixi.js'
    import { useEditorStore } from '../store'
    import type { ViewportCamera } from './ViewportCamera'

    export function setupBoneCreation(app: Application, camera: ViewportCamera): () => void {
      const stage = app.stage

      function onStagePointerDown(e: FederatedPointerEvent): void {
        // Only left-click, only 'select' tool (so gizmo drags don't accidentally create bones)
        if (e.button !== 0) return
        const state = useEditorStore.getState()
        if (state.activeTool !== 'select') return
        if (state.editorMode !== 'pose') return // no bone creation in animate mode (Phase 1)

        // Prevent creating a bone if click was on a bone Graphics (those stop propagation)
        // This handler only fires when click reaches the stage (background click)
        const worldPos = camera.screenToWorld(e.global.x, e.global.y)
        const parentId = state.selectedBoneId // null = root bone

        const newBoneId = state.createBone(parentId)

        // Place the new bone at click position in local space
        const newBone = useEditorStore.getState().skeleton.bones[newBoneId]
        if (parentId) {
          // Child bone: convert world click position to parent's local space
          const { evaluateWorldTransform } = await import('../model/transforms')
          const parentWorld = evaluateWorldTransform(parentId, state.skeleton)
          const cos = Math.cos(-parentWorld.rotation)
          const sin = Math.sin(-parentWorld.rotation)
          const dx = worldPos.x - parentWorld.x
          const dy = worldPos.y - parentWorld.y
          const localX = (cos * dx - sin * dy) / parentWorld.scaleX
          const localY = (sin * dx + cos * dy) / parentWorld.scaleY
          useEditorStore.getState().setBoneTransform(newBoneId, { x: localX, y: localY })
        } else {
          // Root bone: local IS world
          useEditorStore.getState().setBoneTransform(newBoneId, { x: worldPos.x, y: worldPos.y })
        }

        // Select the new bone and switch to Move tool so user can drag to reposition
        useEditorStore.getState().setSelectedBone(newBoneId)
        useEditorStore.getState().setActiveTool('move')
      }

      stage.on('pointerdown', onStagePointerDown)
      return () => stage.off('pointerdown', onStagePointerDown)
    }
    ```
